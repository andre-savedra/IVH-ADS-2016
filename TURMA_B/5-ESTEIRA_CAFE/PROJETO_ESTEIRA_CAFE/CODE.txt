(*The main code inserted by the user and executed associated with the MainTask must be inserted into this POU.*)
PROGRAM UserPrg
VAR
	FEED_MOTOR :BOOL;
	S1_LEVEL_SENSOR : BOOL;
	S2_LEVEL_SENSOR : BOOL;
	S3_PACKAGE_SENSOR : BOOL;
	S4_PACKAGE_COUNTER : BOOL;
	Y1_DOSAGE_VALV : BOOL;
	Y2_DOSAGE_VALV : BOOL;
	Y3_DIRECTION_VALV : BOOL;	
	BELT_MOTOR : BOOL;
	ON_OFF_BUTTON : BOOL;
	EMERGENCY_BUTTON : BOOL;
	EMERGENCY_LAMP : BOOL;

    ON_OFF_BUFFER : BOOL;
    SYSTEM_ON : BOOL;
    DOSAGE_STEPS: INT;
	FEED_STEPS: INT;
	TEST_MODE : BOOL;

	TIMER : TON;
	OPENING_TIME :TIME := T#2.5S;	
	
	COUNTER_BUFFER : BOOL;
	PARTS_COUNTER : INT;
	BATCH_COUNTER : INT;
	PRODUCTION_TARGET : INT;
END_VAR


//MAPA DAS ENTRADAS
IF (NOT TEST_MODE) THEN
	ON_OFF_BUTTON := I00;
	EMERGENCY_BUTTON := NOT I01;
	S1_LEVEL_SENSOR := I02;
	S2_LEVEL_SENSOR := I03;
	S3_PACKAGE_SENSOR := I04;
	S4_PACKAGE_COUNTER := I05;	
END_IF

//MAPA DAS SAÍDAS
Q00 := FEED_MOTOR;
Q01 := Y1_DOSAGE_VALV;
Q02 := Y2_DOSAGE_VALV;
Q03 := Y3_DIRECTION_VALV;
Q04 := BELT_MOTOR;
Q05 := EMERGENCY_LAMP;


// LÓGICA DA BORDAR DE LIGAR E DESLIGAR
IF ( ON_OFF_BUTTON ) THEN
    ON_OFF_BUFFER := TRUE;

ELSIF ( ON_OFF_BUFFER ) THEN
    ON_OFF_BUFFER := FALSE;
    // INVERTE O ESTADO LÓGICO
    SYSTEM_ON := NOT SYSTEM_ON;
END_IF;


// SISTEMA LIGADO
IF ( SYSTEM_ON ) THEN
    // SE EMERGENCIA ACIONADO, DESLIGA TUDO
    IF ( EMERGENCY_BUTTON ) THEN
        SYSTEM_ON := FALSE;
    END_IF;
	
	IF ( S4_PACKAGE_COUNTER ) THEN
    	COUNTER_BUFFER := TRUE;
	ELSIF ( COUNTER_BUFFER ) THEN
		COUNTER_BUFFER := FALSE;
		PARTS_COUNTER := PARTS_COUNTER + 1;
		
		IF(PARTS_COUNTER > BATCH_COUNTER) THEN
			PARTS_COUNTER := 1;
			BATCH_COUNTER := BATCH_COUNTER + 1;
			Y3_DIRECTION_VALV := NOT Y3_DIRECTION_VALV;
			
			IF(BATCH_COUNTER = PRODUCTION_TARGET) THEN
				SYSTEM_ON := FALSE;
			END_IF;
		END_IF;
	END_IF;
	
	
	
	
	CASE FEED_STEPS OF		
		0: // CHECA SE TEM O MÍNIMO DE CAFÉ
		
		IF(S2_LEVEL_SENSOR)THEN
 			FEED_MOTOR := FALSE;			
		ELSE
			FEED_STEPS := 1;			
		END_IF;

		1: // INICIA O ABASTECIMENTO
		
		IF(S1_LEVEL_SENSOR)THEN
 			FEED_STEPS := 0;			
		ELSE
			FEED_MOTOR := TRUE;		
		END_IF;	
	END_CASE;
	
	
	CASE DOSAGE_STEPS OF
		
		0: // GARANTE QUE TEM PACOTE				
		IF(S3_PACKAGE_SENSOR) THEN
			BELT_MOTOR := FALSE;						
			DOSAGE_STEPS := 1;
		// SE NÃO TEM PACOTE, LIGA A ESTEIRA
		ELSE
			Y1_DOSAGE_VALV := FALSE;
    		Y2_DOSAGE_VALV := FALSE;
			BELT_MOTOR := TRUE;
		END_IF;
		
		1: // CHECA SE EXISTE O MÍNIMO DE CAFÉ
		IF(S2_LEVEL_SENSOR) THEN									
			DOSAGE_STEPS := 2;		
		ELSE
			Y1_DOSAGE_VALV := FALSE;
    		Y2_DOSAGE_VALV := FALSE;
			BELT_MOTOR := FALSE;
		END_IF;
		
		
		2: // INICIA A DOSAGEM DO Y1
		3: // INICIA A DOSAGEM DO Y2
		TIMER(IN:=TRUE,PT:=OPENING_TIME);
		
		//ATINGIU O TEMPO DE CONTAGEM
		IF(TIMER.Q) THEN												
			DOSAGE_STEPS := DOSAGE_STEPS + 1;		
		ELSE
			Y1_DOSAGE_VALV := DOSAGE_STEPS = 2;
    		Y2_DOSAGE_VALV := DOSAGE_STEPS = 3;
			BELT_MOTOR := FALSE;
		END_IF;
		
		4: // MANDAR O PACOTE EMBORA (SAIR DO SENSOR S3)
		
		IF(S3_PACKAGE_SENSOR) THEN
			Y1_DOSAGE_VALV := FALSE;
    		Y2_DOSAGE_VALV := FALSE;
			BELT_MOTOR := TRUE;
		ELSE
			BELT_MOTOR := FALSE;
			DOSAGE_STEPS := 0; // REINICIA O PROCESSO
		END_IF;
	END_CASE;
	

// SISTEMA DESLIGADO
ELSE 
    FEED_MOTOR := FALSE;
    Y1_DOSAGE_VALV := FALSE;
    Y2_DOSAGE_VALV := FALSE;
    Y3_DIRECTION_VALV := FALSE;
    BELT_MOTOR := FALSE;
    EMERGENCY_LAMP := EMERGENCY_BUTTON;
END_IF;
